# Babel 插件如何实现 JSX 到 JS 的编译？

module.exports = function(babel) {
  var t = babel.types

  return {
    name: 'custom-jsx-plugin',

    visitor: {
      JSXElement(path) {
        var openingElement = path.node.openingElement

        var tagName = openingElement.name.name

        var args = []
        args.push(t.stringLiteral(tagName))
        var attribs = t.nullLiteral()
        args.push(attribs)
        var reactIdentifier = t.identifier('React') //object

        var createElementIdentifier = t.identifier('createElement')

        var callee = t.memberExpression(
          reactIdentifier,
          createElementIdentifier
        )

        var callExpression = t.callExpression(callee, args)

        callExpression.arguments = callExpression.arguments.concat(
          path.node.children
        )

        path.replaceWith(callExpression, path.node)
      },
    },
  }
}

# 手写发布订阅

class EventListener {
  constructor() {
    this.events = {}
  }
  on(type, fn) {
    if (!type) return
    let fns = this.events[type]
    if (!fns) this.events[type] = []
    if (typeof fn !== 'function') return
    this.events[type].push(fn)
  }
  off(type, fn) {
    if (!type) this.events = {}
    let fns = this.events[type]
    if (!fns) return
    if (typeof fn !== 'function') this.events[type] = []
    fns = fns.filter((e) => e !== fn)
    this.events[type] = fns
  }
  emit(type, ...args) {
    if (!type) this.events = {}
    let fns = this.events[type]
    if (!fns) return
    fns.forEach((fn) => fn.call(this, args))
  }
}

function arr2Tree(arr) {
  let result = []
  if (!Array.isArray(arr)) return result
  let map = {}
  arr = arr.map((e) => {
    let temp = map[e.id]
    if (!temp) {
      temp = []
      map[e.id] = temp
      e.children = temp
    }
    return e
  })
  arr.forEach((item) => {
    let temp = map[item.pid]
    if (temp) {
      temp.push(item)
    } else {
      result.push(item)
    }
  })
  return result
}

# 实现数组读取负数的索引

Proxy 代理实现

let person = ['coderly', 18, '男', '杭州市']
let myProxy = function(ele) {
  return new Proxy(ele, {
    get(target, key, receiver) {
      let index = Number(key)
      let len = target.length
      if (index < 0) {
        index = (len + index) % len
      }
      return target[index]
    },
  })
}
let p = myProxy(person)
console.log('value：', p[-4])
